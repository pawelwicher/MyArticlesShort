module Main where
import System.Random

randomList :: Int -> Int -> IO [Int]
randomList a b = getStdGen >>= return . randomRs (a, b)

main = do
    rndList <- (randomList 0 26)
    let rndString = [(['A' .. 'Z'] ++ [' ']) !! x | x <- rndList]    
    let str = take 10 $ drop 10 rndString 
    print str

-- "AVIULMURWB"
