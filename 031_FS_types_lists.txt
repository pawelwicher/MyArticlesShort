let list0 = []
let list1 = [1]
let list2 = [2; 3]
let list3 = 1 :: list2
let list4 = list1 @ list2
let list5 = [0 .. 2]
let list6 = [5 .. 7]
let list7 = [for i in 1 .. 3 -> i + 1]
let lists = [list0; list1; list2; list3; list4; list5; list6; list7]

lists |> List.iter (fun list -> printfn "%O" list)

// []
// [1]
// [2; 3]
// [1; 2; 3]
// [1; 2; 3]
// [0; 1; 2]
// [5; 6; 7]
// [2; 3; 4]
